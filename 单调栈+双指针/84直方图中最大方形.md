https://leetcode.com/problems/largest-rectangle-in-histogram/


Example:

Input: [2,1,5,6,2,3]  
Output: 10  

```
class Solution:
    def largestRectangleArea(self, heights: List[int]) -> int:
        stack = []
        index,ans = 0,0
        n = len(heights)
        while(index < n):
            if len(stack)== 0 or stack[-1]<= heights[index]:
                stack.append(index)
                index = index+1
            else:
                up = heights[stack[-1]]
                stack.pop()
                #up = heights[stack[-1]]
                if len(stack)!=0:#弹出后栈仍不为kong
                    ans = max(ans,(index-stack[-1]-1)*up)
                else:
                    ans = max(ans,up*index)
                stack.append(index)
                index = index+1
        #栈维护完了后，因为可能栈里还有一堆单调递增的序列，需要计算
        while (len(stack) != 0):
            height = heights[stack[-1]]
            stack.pop()
            #height = stack[i]
            if len(stack)== 0:
                ans =max(ans,n*height)
            else:
                ans = max(ans,height *(n-stack[-1]-1)) 
        return ans
```
